    % Complex Numbers (outer square)

    \draw[thick, fill=yellow!40] (-7,0) ellipse (4.5cm and 3cm);

    \node at (-9.5, 0) {\Large$\mathbb{Z}$};

    \draw[thick, fill=green!40] (1,0) ellipse (2cm and 1.2cm);
    \node at (1, 0) {\Large$\mathbb{R} /\mathbb{Q} $};
